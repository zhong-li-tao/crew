# 项目结构

项目根目录/
├── main.py                    # 主程序入口
├── test.py                    # RAGAS评估测试
├── model_download.py          # 模型下载工具
├── 数据/
│   └── 员工手册/
│       ├── 员工手册.json      # 结构化数据（自动生成的）
│       ├── 员工手册.pdf       # 原始PDF（自动下载）
│       ├── 员工手册.txt       # 文本内容（自动解析）
│       ├── gain_data.py       # PDF爬取与解析脚本
│       └── convert_to_json.py # JSON转换脚本
├── 知识库/
│   ├── chunk.py               # 文本分块模块
│   ├── embedding.py           # 向量化模块
│   ├── vector_db.py           # 向量数据库模块
│   └── utils.py               # 工具类（rag类）
└── 检索层/
    └── retriever.py           # 检索模块
